import numpy as np

def grossman_model(rho, a, C, N):
    P_true = 100
    signals = P_true + np.random.normal(0, (1-rho)*5, N)
    Var_post = (1-rho**2)*25
    X = (signals - P_true) / (a*Var_post)
    X = X * (1 - C)  # info cost dampening
    P_eq = np.mean(signals - a*Var_post*X)
    info_index = rho * (1 - C)
    return X, P_eq, info_index
